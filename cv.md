function duplicateEncode(word){
  const arr = word.toLowerCase().split('');

  let dupli = arr.reduce((acc, item) => {
    acc[item] = acc[item] ? acc[item] + 1 : 1;
    return acc;
  }, {});

  let result = [];

  arr.forEach(item => {
    if (dupli[item] === 1) {
      result.push('(');
    } else {
      result.push(')');
    }
  });

  return result.join('');
}
